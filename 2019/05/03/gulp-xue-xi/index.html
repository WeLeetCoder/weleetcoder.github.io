var gulp = require('gulp');
var sass = require('gulp-sass');
var del = require('del');
var sourceMap = require('gulp-sourcemaps');
var autoPrefixer = require('gulp-autoprefixer');

gulp.task('clean', function() {
  return del.sync(["./src/css/*.css"]);
})

gulp.task("sass", function() {
  return (
    gulp
      .src(["./src/sass/**/*.scss", "./src/sass/**/*.sass"])

      .pipe(sass({ outputStyle: "expanded" }).on("error", sass.logError))
      .pipe(sourceMap.init())
      .pipe(autoPrefixer())
      // .pipe(sass({outputStyle: 'expanded'}).on("error", sass.logError))
      .pipe(sourceMap.write("."))
      .pipe(gulp.dest("./src/css"))
  );
});

gulp.task('sass:w', function() {
  gulp.watch(
    ["./src/sass/**/*.scss", "./src/sass/**/*.sass"],
    gulp.series("sass")
  );
});

gulp.task('p', function() {
  return gulp
    .src("./src/css/*.css")

    .pipe(gulp.dest("./src/css"));
});

gulp.task("default", gulp.series("clean", "sass"));

Gulp API

gulp 为我们提供了许多便利的 API 供我们操作文件流。

gulp.src

gulp.src(globs[, options])

读取并输出所有与给定的 glob（文件匹配符） 或 glob 数组相匹配的文件，返回一个 vinyl文件 的流，以便通过管道（pipe）传送给插件处理。

globs 参数

Type: String or Array

需要读取的字符串 glob 或 glob 数组。

gulp.dest(path[, options])
